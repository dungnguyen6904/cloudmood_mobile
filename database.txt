CREATE TABLE public.User (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  fullName character varying NOT NULL,
  email character varying NOT NULL,
  password character varying NOT NULL,
  avatar character varying,
  role boolean NOT NULL DEFAULT true,
  createdAt date NOT NULL,
  CONSTRAINT User_pkey PRIMARY KEY (id)
);
CREATE TABLE public.Category (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  name character varying NOT NULL,
  CONSTRAINT Category_pkey PRIMARY KEY (id)
);
CREATE TABLE public.Place (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  name character varying NOT NULL,
  description character varying NOT NULL,
  latitude double precision NOT NULL,
  longitude double precision NOT NULL,
  address character varying NOT NULL,
  price character varying NOT NULL,
  openTime time with time zone NOT NULL,
  closeTime time with time zone NOT NULL,
  categoryId bigint NOT NULL,
  image character varying NOT NULL,
  rating double precision,
  userRatingCount integer,
  externalId character varying UNIQUE,
  phone character varying,
  website character varying,
  priceLevel character varying,
  amenities ARRAY,
  lastSyncedAt timestamp without time zone DEFAULT now(),
  CONSTRAINT Place_pkey PRIMARY KEY (id),
  CONSTRAINT fk_place_category FOREIGN KEY (categoryId) REFERENCES public.Category(id)
);
CREATE TABLE public.Review (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  rating real NOT NULL,
  comment character varying NOT NULL,
  userId bigint,
  placeId bigint NOT NULL,
  externalReviewId character varying UNIQUE,
  authorName character varying,
  authorAvatar character varying,
  authorLocation character varying,
  publishedDate date,
  source character varying DEFAULT 'LOCAL'::character varying,
  CONSTRAINT Review_pkey PRIMARY KEY (id),
  CONSTRAINT fk_review_place FOREIGN KEY (placeId) REFERENCES public.Place(id),
  CONSTRAINT fk_review_user FOREIGN KEY (userId) REFERENCES public.User(id)
);
CREATE TABLE public.Itinerary (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  title character varying NOT NULL,
  startDate date NOT NULL,
  days bigint,
  budget bigint,
  userId bigint NOT NULL,
  CONSTRAINT Itinerary_pkey PRIMARY KEY (id),
  CONSTRAINT fk_itinerary_user FOREIGN KEY (userId) REFERENCES public.User(id)
);
CREATE TABLE public.ItineraryDetail (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  arrivalTime time with time zone NOT NULL,
  leaveTime time with time zone NOT NULL,
  order character varying NOT NULL,
  placeId bigint NOT NULL,
  itineraryId bigint NOT NULL,
  CONSTRAINT ItineraryDetail_pkey PRIMARY KEY (id),
  CONSTRAINT fk_detail_itinerary FOREIGN KEY (itineraryId) REFERENCES public.Itinerary(id),
  CONSTRAINT fk_detail_place FOREIGN KEY (placeId) REFERENCES public.Place(id)
);
CREATE TABLE public.PlacePhoto (
  id bigint NOT NULL DEFAULT nextval('"PlacePhoto_id_seq"'::regclass),
  placeId bigint NOT NULL,
  urlOriginal character varying NOT NULL,
  urlThumbnail character varying,
  caption text,
  source character varying DEFAULT 'TRIPADVISOR'::character varying,
  CONSTRAINT PlacePhoto_pkey PRIMARY KEY (id),
  CONSTRAINT PlacePhoto_placeId_fkey FOREIGN KEY (placeId) REFERENCES public.Place(id)
);
